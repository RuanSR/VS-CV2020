
CREATE TABLE Atendente(
	[id_atendente] [int] PRIMARY KEY IDENTITY(1,1) NOT NULL,
	[usuario_atendente] [varchar](30) NOT NULL,
	[senha_atendente] [nvarchar](10) NOT NULL
)

CREATE TABLE Cliente(
	[id_cliente] [int] PRIMARY KEY IDENTITY(1,1) NOT NULL,
	[nome_cliente] [varchar](50) NOT NULL,
	[apelido_cliente] [varchar](20) NOT NULL,
	[endereco_cliente] [varchar](100) NOT NULL,
	[telefone_cliente] [varchar](16) NOT NULL,
	[cpf_cliente] [varchar](14) NOT NULL,
	[limite_conta_cliente] [varchar](20) NOT NULL,
	[total_conta_cliente] [varchar](20) NOT NULL,
	[total_pago_cliente] [varchar](20) NOT NULL,
	[num_notas_cliente] [int] NOT NULL,
	[data_conta_cliente] [varchar](10) NOT NULL,
	[status_cliente] [bit] NOT NULL,
)

CREATE TABLE Registro(
	[id_registro] [int] PRIMARY KEY IDENTITY(1,1) NOT NULL,
	[data_nota_registro] [varchar](8) NOT NULL,
	[hora_nota_registro] [varchar](8) NOT NULL,
	[atendente_registro] [varchar](50) NULL,
	[valor_registro] [varchar](20) NOT NULL,
	[log_registro] [text] NOT NULL,
	[id_cliente_registro] [int] NOT NULL
)
